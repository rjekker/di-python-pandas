Counting misses

Create a class that counts the number of times you tried to access an attribute that does not exist.

The class also has a default value; when you try to access a non-existing property, it returns that default.

Some example code 

class CountMisses:
    def __init__(self, name, data, default_value=None):
        self.name = name
        self.data = data
        self.default_value = default_value
        self.__misses = 0

    def __getattr__(self, attr):
        ... # your code here

obj = CountMisses("my object", [1,2,3,4,5], "*My default value*")
print(obj.name, obj.data, obj.other, obj.glorp, obj.foo)
print(obj)

Example output: 

my object [1, 2, 3, 4, 5] *My default value* *My default value* *My default value*
misses: 3

class CountMisses:
    def __init__(self, name, data, default_value=None):
        self.name = name
        self.data = data
        self.default_value = default_value
        self.__misses = 0

    def __getattr__(self, attr):
        self.__misses += 1
        return self.default_value

    def __str__(self):
        return f"misses: {self.__misses}"

obj = CountMisses("my object", [1,2,3,4,5], "*My default value*")

print(obj.name, obj.data, obj.other, obj.glorp, obj.foo)
print(obj)

Counting hits

Can you extend the CountMisses class so that it also counts succesfull attribute access? You need a trick to get this right, see: This article

Can you explain the number of hits you see in the output?

Solution with an int: this gives a surprising result! 

class CountMisses:
    def __init__(self, name, data, default_value=None):
        self.name = name
        self.data = data
        self.default_value = default_value
        self._misses = 0
        self._hits = 1

    def __getattribute__(self, attr):
        if attr in super().__getattribute__("__dict__"):
            current_hits = super().__getattribute__("_hits")
            super().__setattr__("_hits", current_hits+1)
        return super().__getattribute__(attr)

    def __getattr__(self, attr):
        self._misses += 1
        return self.default_value

    def __str__(self):
        return f"hits: {self._hits}, misses: {self._misses}"

obj = CountMisses("my object", [1,2,3,4,5], "*My default value*")

print(obj.name, obj.data, obj.other, obj.glorp, obj.foo)
print(obj)

Solution with a counter: 

from collections import Counter

class CountMisses:
    def __init__(self, name, data, default_value=None):
        self.name = name
        self.data = data
        self.default_value = default_value
        self._misses = 0
        self._hits = Counter()

    def __getattribute__(self, attr):
        if attr in super().__getattribute__("__dict__"):
            super().__getattribute__("_hits")[attr] += 1
        return super().__getattribute__(attr)

    def __getattr__(self, attr):
        self._misses += 1
        return self.default_value

    def __str__(self):
        return f"hits: {self._hits}, misses: {self._misses}"

obj = CountMisses("my object", [1,2,3,4,5], "*My default value*")

print(obj.name, obj.data, obj.other, obj.glorp, obj.foo)
print(obj)

Scala tuples

In Scala, tuples are created really similarly to Python: 

val a_tuple = ("z", 3, “Python”, -1)

But they’re accessed in a different way: 

println(a_tuple._1) // “z”
println(a_tuple._3) // “Python”

Each element in the tuple is accessed as an attribute, with the first element being the attribute _1, the second _2, and so on.

Can you extend the python class tuple to match this behaviour? 

class Tuple(tuple):
    def __getattr__(self, name):
        def _int(val):
            try:
                return int(val)
            except ValueError:
                return False

        if not name.startswith('_') or not _int(name[1:]):
            raise AttributeError("'tuple' object has no attribute '%s'" % name)
        index = _int(name[1:]) - 1
        return self[index]


t = Tuple(['z', 3, 'Python', -1])
print(t._1)  # 'z'
print(t._2)  # 3
print(t._3)  # 'Python'

Loud

Create a class decorator called loud() that makes sure that, whenever you access an attribute that is a string, you add an exclamation mark at the end of that string?

Example usage: 

@loud
class MyClass:
    def __init__(self):
        self.x = 5
        self.s = "hi"

obj = MyClass()
print(obj.x, obj.s)

output: 

5 hi!

def loud(clz):
    class NewClz(clz):
        def __getattribute__(self, attr):
            val = super().__getattribute__(attr)
            if isinstance(val, str):
                return val + "!"
            return val
    return NewClz

@loud
class MyClass:
    def __init__(self):
        self.x = 5
        self.s = "hi"

obj = MyClass()
print(obj.x, obj.s)
