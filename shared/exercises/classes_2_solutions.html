BankAccount

Given the Bankaccount class:

  • add __str__(), __repr__() methods
  • Add an =__eq__()__ method to the BankAccount class, such that two accounts are equal only if their balances AND owner names are equal. Make sure to test that your code works correctly!
  • Add a __mul__() method to the BankAccount class: this should create a new BankAccount with the names of both accounts added together, and the balances multiplied.
class BankAccount:
    def __init__(self, name, initial_balance):
        self.name = name
        self.balance = initial_balance

    def deposit(self, amount):
        if amount > 0:
            self.balance += amount
            return self.balance
        else:
            print("Can't deposit a negative amount!")

    def withdraw(self, amount):
        if amount > 0:
            if amount <= self.balance:
                self.balance -= amount
                return self.balance
            else:
                print("Insufficient balance!")
        else:
            print("Can't withdraw nonpositive amount!")

    def __str__(self):
        return "Account for: " + self.name + ", balance: " + str(self.balance)

    def __repr__(self):
        return self.__str__()

    def __eq__(self, other):
        return self.name == other.name and self.balance == other.balance

    def __mul__(self, other):
        return BankAccount(self.name+other.name, self.balance*other.balance)

Vectors

class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        self.len = (x**2 + y**2)**0.5

    def __repr__(self):
        return "({},{})".format(self.x, self.y)

    def __abs__(self):
         return (self.x**2 + self.y**2)**0.5

    def __lt__(self, other):
        return self.len < other.len

    def __add__(self, other):
        return Vector(self.x + other.x, self.y + other.y)

Weird integers

Create a "WeirdInt" class which inherits from int and redefines certain methods:

  • make it so + usually gives the right answer, but not always (use the random module)
  • make it so the str() version of a WeirdInt is something odd, e.g., str(3) return 'three', but str() of other numbers returns the number with some leading and trailing spaces
from random import choice
class WeirdInt(int):
    def __add__(self, other):
        if choice([False, False, False, False, True]): # 1 in 5 chance
            return WeirdInt(-1 * (self-other))
        else:
            return WeirdInt(super().__add__(other))

    def __str__(self):
        if self == 3:
            return "three"
        else:
            return " "*self + super().__str__() + " "*self
