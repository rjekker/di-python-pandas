Exercise

Use flask-restful to serve the data from your database. Supported methods should be:

GET the entire table POST to create a new object GET a specific object by id PUT to update an object DELETE an object

Create a separate client using requests to test these methods. 

from flask import Flask, render_template, jsonify, request
from flask_sqlalchemy import SQLAlchemy
from flask_restful import reqparse, abort, Api, Resource

app = Flask(__name__)
api = Api(app)

app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///users.db'
db = SQLAlchemy(app)


class User(db.Model):
   id = db.Column(db.Integer, primary_key=True)
   username = db.Column(db.String(80), unique=True, nullable=False)
   email = db.Column(db.String(120), unique=True, nullable=False)

   def __repr__(self):
       return '<User %r>' % self.username


# Please implement another resource
# To support GET, PUT and DELETE for single instances
class UserApi(Resource):
   def get(self, id):
       user = User.query.filter(User.id == id).first()
       return jsonify({"username": user.username, "email": user.email})

   def put(self, id):
       data = request.get_json(force=True)
       user = User.query.filter(User.id == id).first()
       user.username = data['username']
       user.email = data['email']
       db.session.add(user)
       db.session.commit()
       return {"username": user.username, "email": user.email}, 201

   def delete(self, id):
       user = User.query.filter(User.id == id).first()
       db.session.delete(user)
       db.session.commit()
       return '', 204


class UserListApi(Resource):
    def get(self):
        users = User.query.all()
        return jsonify([dict(id=u.id, username=u.username, email=u.email) for u in users])

    def post(self):
        user = User(**request.get_json(force=True))
        db.session.add(user)
        db.session.commit()
        return {"id": user.id, "username": user.username, "email": user.email}, 201


api.add_resource(UserApi, '/api/users/<int:id>')
api.add_resource(UserListApi, '/api/users')
