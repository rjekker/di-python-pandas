Countries

Start by exploring the API at https://restcountries.eu/

Ask the user to enter the name of a country and retrieve the capital for that country.

When there is no matching country, print an appropriate message.

If the user enters only a part of the name of a country, you get multiple matches. Print the country names with capitals for all matches. 

import requests

country = input("Country? ")
resp = requests.get("https://restcountries.com/v3.1/name/" + country)

if resp.status_code == 404:
    print("No such country")
else:
    data = resp.json()
    for country in data:
        name = country['name']['official']
        if "capital" in country:
            print(name, " - with capital: ", country["capital"])
        else:
            print(f"{name} has no capital")

Borders

Each country has a list of bordering countries, called "borders".

  • Ask the user to input a country name. Print the names of the bordering countries
import requests

country = input("Country? ")
resp = requests.get("https://restcountries.com/v3.1/name/" + country)

if resp.status_code == 404:
    print("No such country")
    exit(1)


for country in resp.json():
    name = country['name']['official']
    print(f"Borders for {name}:")
    print(f"-------------------------")
    if "borders" in country:
        borders = country["borders"]
        for b in borders:
            resp = requests.get("https://restcountries.com/v3.1/alpha/" + b)
            print(resp.json()[0]["name"]["official"])
    else:
        print(f"{name} has no borders ??")
    print()

Cars and number plates

How much do all purple lamborghini's in the Netherlands weigh together? 

import requests

brand = input("Which brand? ").upper()
color = input("Which color (in Dutch)? ").upper()
resp = requests.get(f"https://opendata.rdw.nl/resource/m9d7-ebf2.json?merk={brand}&eerste_kleur={color}")

if resp.status_code != 200:
    print(f"Received status code {resp.status_code}")
    exit(1)

data = resp.json()

if not data:
    print(f"No cars of brand {brand} and color {color}")
    exit(2)

total_mass = 0
for auto in data:
    if "massa_rijklaar" in auto:
        total_mass += float(auto["massa_rijklaar"])

print("Total weight:", total_mass)

Fuel

Ask the user for a number plate. Then print the type of fuel this car uses.

Example number plate: G-922-RK 

import requests

kenteken = input("Numberplate? ").replace("-", "").upper()
resp = requests.get(f"https://opendata.rdw.nl/resource/m9d7-ebf2.json?kenteken={kenteken}")

if resp.status_code != 200:
    print(f"Received status code {resp.status_code}")
    exit(1)

data = resp.json()

if not data:
    print(f"No cars with kenteken {kenteken}")
    exit(2)

try:
    fuel_url = data[0]["api_gekentekende_voertuigen_brandstof"]
    resp = requests.get(fuel_url + f"?kenteken={kenteken}")
    if resp.status_code != 200:
        print(f"Received status code {resp.status_code}")
        exit(1)
    print(resp.json()[0]['brandstof_omschrijving'])
except KeyError:
    print("Cannot get fuel..")

Star wars

For this exercise you can use http://swapi.dev/api

Write a program that asks for the name of a Star Wars character.

Print:

  • The name of their planet of origin
  • The title of every movie the character appears in

A movie class

import requests


class Movie:
    BASE_URL = "https://swapi.dev/api/films"

    def __init__(self, id):
        """All swapi objects have an id"""
        self.id = id
        self._data = None
        self.url = f"{Movie.BASE_URL}/{self.id}"

    def get_data(self):
        response = requests.get(self.url)
        if response.status_code != 200:
            raise ValueError("Cannot retrieve data for", self.url)
        self._data = response.json()

    def get_title(self):
        if not self._data:
            self.get_data()
        return self._data['title']


m = Movie(3)
print(m.get_title())

Movie Characters

import requests

class Movie:
    BASE_URL = "https://swapi.dev/api/films"

    def __init__(self, id):
        """All swapi objects have an id"""
        self.id = id
        self._data = None
        self.url = f"{Movie.BASE_URL}/{self.id}"

    def get_data(self):
        response = requests.get(self.url)
        if response.status_code != 200:
            raise ValueError("Cannot retrieve data for", self.url)
        self._data = response.json()

    def get_title(self):
        if not self._data:
            self.get_data()
        return self._data['title']

    def get_single_character(self, url):
        response = requests.get(url)
        data = response.json()
        return data['name']

    def get_characters(self):
        return [self.get_single_character(c) for c in self._data["characters"]]


m = Movie(3)
print(m.get_title())
for c in m.get_characters():
    print(c)
